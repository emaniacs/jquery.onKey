var Txt1Handler = {
    'a': function(e) {
        var special = [];
        e.shiftKey ? special.push('shift') : '';
        e.altKey ? special.push('alt') : '';
        e.ctrlKey ? special.push('ctrl') : '';
        e.metaKey ? special.push('meta') : '';
        
        $('.txtOutput1').html(special.join('+') + ' + a');
    },
    '/': function(e) {
        e.preventDefault();
        $('.txtOutput1').html('Value is: '+ $(this).val());
    }
};

var Txt2Handler = {
    'a' : function(e) {
        var key = e.which || e.keyCode;
        $('.txtOutput2').html(key);
    },
    'unknown': function(e){
        $('.txtOutput2').html('No handler for ('+(e.which||e.keyCode)+')');
    },
    'enter': function(e) {
        e.preventDefault();
        $('.txtOutput2').html('You press Enter');
    }
};

Txt2Handler['tab'] = function(e) {
    console.log('remove enter.');
    Txt2Handler['enter'] = undefined;
};

$(document).ready(function(){
    $('.txt1').onKey(Txt1Handler);
    $('.txt2').onKey(Txt2Handler);
});
